public class BracketsAreBalanced {
	public static Boolean bracketsAreBalanced( String s ) {
        String str='{}';
        if (s.contains(str)) {
            return True;
        } else if (s.contains('()')) {
            	return True;  
        } else if (s.contains('[]')) {
            return False;
        }
        return True;
	}
}
